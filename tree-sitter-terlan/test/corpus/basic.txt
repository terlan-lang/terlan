module web.Routes.

import std.http.Request.
import std.http.Response.

@route {
  method = "GET"
  path = "/users/:id"
}
pub show_user(request: Request): Response ->
  Response.text("ok", status = 200).

pub (user: User) display_name(): String ->
  case user.name {
    "" -> "guest";
    _ -> user.name
  }.

pub (user: User) email(): String ->
  user.#email.

pub choose(user: User): String ->
  if {
    True -> user.display_name();
    _ -> "unknown"
  }.

pub lookup(): Query ->
  sql {
    SELECT id, name FROM users WHERE id = $1
  }.
